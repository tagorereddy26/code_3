import numpy as np
import matplotlib.pyplot as plt
from tensorflow.keras.datasets import reuters
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense
from tensorflow.keras.utils import to_categorical

# Load dataset
(x_train, y_train), (x_test, y_test) = reuters.load_data(num_words=10000)

# Vectorize sequences
def vectorize(seqs, dim=10000):
    x = np.zeros((len(seqs), dim))
    for i, s in enumerate(seqs):
        x[i, s] = 1
    return x

x_train = vectorize(x_train)
x_test = vectorize(x_test)

# One-hot encode labels
y_train = to_categorical(y_train, 46)
y_test = to_categorical(y_test, 46)

# Validation split
x_val, partial_x_train = x_train[:1000], x_train[1000:]
y_val, partial_y_train = y_train[:1000], y_train[1000:]

# Build model
model = Sequential([
    Dense(64, activation='relu', input_shape=(10000,)),
    Dense(64, activation='relu'),
    Dense(46, activation='softmax')
])

model.compile(optimizer='rmsprop',
              loss='categorical_crossentropy',
              metrics=['accuracy'])

# Train model
history = model.fit(partial_x_train, partial_y_train,
                    epochs=20, batch_size=512,
                    validation_data=(x_val, y_val))

# Plot loss
plt.plot(history.history['loss'], 'bo', label='Training loss')
plt.plot(history.history['val_loss'], 'b', label='Validation loss')
plt.legend()
plt.show()

# Retrain with optimal epochs
model.fit(partial_x_train, partial_y_train,
          epochs=8, batch_size=512,
          validation_data=(x_val, y_val))

# Evaluate
loss, acc = model.evaluate(x_test, y_test)
print("Test loss, Test accuracy:", [loss, acc])
